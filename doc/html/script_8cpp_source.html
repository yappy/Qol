222 {
223  std::vector<std::string> result;
224 
225  int tind = lua_absindex(L, ind);
226 
227  int type = lua_type(L, ind);
228  if (type == LUA_TNONE) {
229  throw std::logic_error("invalid index: " + std::to_string(ind));
230  }
231  const char *typestr = lua_typename(L, type);
232  // copy and tostring it
233  lua_pushvalue(L, ind);
234  const char *valstr = lua_tostring(L, -1);
235  valstr = (valstr == NULL) ? "" : valstr;
236  // pop copy
237  lua_pop(L, 1);
238  // boolean cannot be tostring
239  if (type == LUA_TBOOLEAN) {
240  valstr = lua_toboolean(L, ind) ? "true" : "false";
241  }
242 
243  {
244  std::string str;
245  for (int i = 0; i < depth; i++) {
246  str += " ";
247  }
248  str += "(";
249  str += typestr;
250  str += ") ";
251  str += valstr;
252  result.emplace_back(std::move(str));
253  }
254  if (type == LUA_TTABLE && depth < maxDepth) {
255  lua_pushnil(L);
256  while (lua_next(L, tind) != 0) {

258  auto key = luaValueToStrList(L, -2, maxDepth, depth + 1);
259  auto val = luaValueToStrList(L, -1, maxDepth, depth + 1);
260  result.insert(result.end(), key.begin(), key.end());
261  result.insert(result.end(), val.begin(), val.end());
