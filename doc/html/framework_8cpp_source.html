272 void FrameControl::endFrame()
273 {
274  // for fps calc
275  if (!shouldSkipFrame()) {
276  m_fpsFrameAcc++;
277  }
278 
279  int64_t target = m_base + m_counterPerFrame;
280  int64_t cur = getTimeCounter();
281  if (m_base == 0) {
282  // force OK
283  m_base = cur;
284  }
285  else if (cur < target) {
286  // OK, wait for next frame
287  while (cur < target) {
288  ::Sleep(1);
289  cur = getTimeCounter();
290  }
291  // may overrun a little bit, add it to next frame
292  m_base = target;
293  }
294  else {
295  // too late, probably immediately right after vsync
296  m_base = cur;
297  }
298 
299  // m_frameCount++;
300  // m_frameCount %= (#draw(=1) + #skip)
301  m_frameCount = (m_frameCount + 1) % (1 + m_skipCount);
302 
303  // for fps calc
304  m_fpsCount++;
305  if (m_fpsCount >= m_fpsPeriod) {
306  double sec = static_cast<double>(cur - m_fpsBase) / m_freq;
307  m_fps = m_fpsFrameAcc / sec;
308 
309  m_fpsCount = 0;
310  m_fpsFrameAcc = 0;
311  m_fpsBase = cur;
312  }
313 }
