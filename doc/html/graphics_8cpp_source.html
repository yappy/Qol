40 FrameControl::FrameControl(int64_t fpsNumer, int64_t fpsDenom) :
41  m_base(0), m_skipCount(0),
42  m_fpsPeriod(static_cast<int>(fpsNumer / fpsDenom))
43 {

46  BOOL b = ::QueryPerformanceFrequency(&freq);

51  // = m_freq / (fpsNumer / fpsDenom)
52  // = m_freq * fpsDenom / fpsNumer
53  m_countPerFrame = m_freq * fpsDenom / fpsNumer;
54 }

61 void FrameControl::endFrame()
62 {
63  if (shouldSkipFrame()) {
64  m_fpsSkipAcc++;
65  }
66  else {
67  m_fpsFrameAcc++;
68  }
69 
70  int64_t target = m_base + m_countPerFrame * (m_skipCount + 1);
71  int64_t cur = getTimeCounter();
72  // cur < target OK
73  // m_base == 0 the first frame, force OK
74  // m_skipCount > MaxSkipCount force OK
75  if (cur < target || m_base == 0 || m_skipCount > MaxSkipCount) {
76  // OK, wait for next frame
77  while (cur < target && m_base != 0 && m_skipCount <= MaxSkipCount) {
78  cur = getTimeCounter();
79  ::Sleep(1);
80  }
81  m_base = cur;
82  m_skipCount = 0;
83  }
84  else {
85  // too late, skip the next frame
86  m_skipCount++;
87  }
88 
89  m_fpsCount++;
90  if (m_fpsCount >= m_fpsPeriod) {
91  double sec = static_cast<double>(cur - m_fpsBase) / m_freq;
92  m_fps = m_fpsFrameAcc / sec;
93  m_sps = m_fpsSkipAcc;
94 
95  m_fpsCount = 0;
96  m_fpsFrameAcc = 0;
97  m_fpsSkipAcc = 0;
98  m_fpsBase = cur;
99 
100  debug::writef(L"fps=%.2f (%d)", getFramePerSec(), getSkipPerSec());
101  }
102 }

145 inline void createCBFromTask(CBChanges *out, const DrawTask &task)
146 {
147  // ax + by + cz + d = 0
148  // x = 0.5
149  XMVECTORF32 xMirror = { 1.0f, 0.0f, 0.0f, -0.5f };
150  // y = 0.5
151  XMVECTORF32 yMirror = { 0.0f, 1.0f, 0.0f, -0.5f };
152  out->lrInv = XMMatrixTranspose(
153  task.lrInv ? XMMatrixReflect(xMirror) : XMMatrixIdentity());
154  out->udInv = XMMatrixTranspose(
155  task.udInv ? XMMatrixReflect(yMirror) : XMMatrixIdentity());
156  out->DestScale = XMMatrixTranspose(XMMatrixScaling(
157  static_cast<float>(task.sw), static_cast<float>(task.sh), 1.0f));
158  out->Centering = XMMatrixTranspose(XMMatrixTranslation(
159  -static_cast<float>(task.cx), -static_cast<float>(task.cy), 0.0f));
160  out->Scale = XMMatrixTranspose(XMMatrixScaling(
161  task.scaleX, task.scaleY, 1.0f));
162  out->Rotation = XMMatrixTranspose(XMMatrixRotationZ(task.angle));
163  out->Translate = XMMatrixTranspose(XMMatrixTranslation(
164  static_cast<float>(task.dx), static_cast<float>(task.dy), 1.0f));
165  out->uvOffset = XMFLOAT2(
166  static_cast<float>(task.sx) / task.texture->w,
167  static_cast<float>(task.sy) / task.texture->h);
168  out->uvSize = XMFLOAT2(
169  static_cast<float>(task.sw) / task.texture->w,
170  static_cast<float>(task.sh) / task.texture->h);
171  out->Alpha = task.alpha;
172 }
