47 inline void createCBFromTask(CBChanges *out, const DrawTask &task)
48 {
49  // ax + by + cz + d = 0
50  // x = 0.5
51  XMVECTORF32 xMirror = { 1.0f, 0.0f, 0.0f, -0.5f };
52  // y = 0.5
53  XMVECTORF32 yMirror = { 0.0f, 1.0f, 0.0f, -0.5f };
54  out->lrInv = XMMatrixTranspose(
55  task.lrInv ? XMMatrixReflect(xMirror) : XMMatrixIdentity());
56  out->udInv = XMMatrixTranspose(
57  task.udInv ? XMMatrixReflect(yMirror) : XMMatrixIdentity());
58  out->DestScale = XMMatrixTranspose(XMMatrixScaling(
59  static_cast<float>(task.sw), static_cast<float>(task.sh), 1.0f));
60  out->Centering = XMMatrixTranspose(XMMatrixTranslation(
61  -static_cast<float>(task.cx), -static_cast<float>(task.cy), 0.0f));
62  out->Scale = XMMatrixTranspose(XMMatrixScaling(
63  task.scaleX, task.scaleY, 1.0f));
64  out->Rotation = XMMatrixTranspose(XMMatrixRotationZ(task.angle));
65  out->Translate = XMMatrixTranspose(XMMatrixTranslation(
66  static_cast<float>(task.dx), static_cast<float>(task.dy), 1.0f));
67  out->uvOffset = XMFLOAT2(
68  static_cast<float>(task.sx) / task.texW,
69  static_cast<float>(task.sy) / task.texH);
70  out->uvSize = XMFLOAT2(
71  static_cast<float>(task.sw) / task.texW,
72  static_cast<float>(task.sh) / task.texH);
73  out->FontColor = XMFLOAT4(
74  ((task.fontColor & 0x00ff0000) >> 16) / 255.0f,
75  ((task.fontColor & 0x0000ff00) >> 8) / 255.0f,
76  ((task.fontColor & 0x000000ff) >> 0) / 255.0f,
77  ((task.fontColor & 0xff000000) >> 24) / 255.0f);

460  m_pContext->OMSetBlendState(m_pBlendState.get(), blendFactor, 0xffffffff);
461

464  // Set constant buffer
